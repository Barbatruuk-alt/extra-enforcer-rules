47  /**
48   * Enforcer rule that will check the bytecode version of each class of each dependency.
49   * 
50   * @see <a href="http://en.wikipedia.org/wiki/Java_class_file#General_layout">Java class file general layout</a>
51   * @since 1.0-alpha-4
52   */
53  public class EnforceBytecodeVersion extends AbstractResolveDependencies
54  {
55      private static final Map<String, Integer> JDK_TO_MAJOR_VERSION_NUMBER_MAPPING = new HashMap<String, Integer>();
56      static
57      {
58          JDK_TO_MAJOR_VERSION_NUMBER_MAPPING.put( "1.1", 45 );
59          JDK_TO_MAJOR_VERSION_NUMBER_MAPPING.put( "1.2", 46 );
60          JDK_TO_MAJOR_VERSION_NUMBER_MAPPING.put( "1.3", 47 );
61          JDK_TO_MAJOR_VERSION_NUMBER_MAPPING.put( "1.4", 48 );
62          JDK_TO_MAJOR_VERSION_NUMBER_MAPPING.put( "1.5", 49 );
63          JDK_TO_MAJOR_VERSION_NUMBER_MAPPING.put( "1.6", 50 );
64          JDK_TO_MAJOR_VERSION_NUMBER_MAPPING.put( "1.7", 51 );
65          JDK_TO_MAJOR_VERSION_NUMBER_MAPPING.put( "1.8", 52 );
66      }
67  
68      static String renderVersion( int major, int minor )
69      {
70          if ( minor == 0 )
71          {
72              for ( Map.Entry<String, Integer> entry : JDK_TO_MAJOR_VERSION_NUMBER_MAPPING.entrySet() )
73              {
74                  if ( major == entry.getValue() )
75                  {
76                      return "JDK " + entry.getKey();
77                  }
78              }
79          }
80          return major + "." + minor;
81      }

125     @Override
126     protected void handleArtifacts( Set<Artifact> artifacts )
127         throws EnforcerRuleException
128     {
129         computeParameters();
130 
131         // look for banned dependencies
132         Set<Artifact> foundExcludes = checkDependencies( filterArtifacts( artifacts ), getLog() );
133 
134         // if any are found, fail the check but list all of them
135         if ( foundExcludes != null && !foundExcludes.isEmpty() )
136         {
137             StringBuilder buf = new StringBuilder();
138             if ( message != null )
139             {
140                 buf.append( message + "\n" );
141             }
142             for ( Artifact artifact : foundExcludes )
143             {
144                 buf.append( getErrorMessage( artifact ) );
145             }
146             message = buf.toString() + "Use 'mvn dependency:tree' to locate the source of the banned dependencies.";
147 
148             throw new EnforcerRuleException( message );
149         }
150     }

163     private void computeParameters()
164         throws EnforcerRuleException
165     {
166         if ( maxJdkVersion != null && maxJavaMajorVersionNumber != -1 )
167         {
168             throw new IllegalArgumentException( "Only maxJdkVersion or maxJavaMajorVersionNumber "
169                 + "configuration parameters should be set. Not both." );
170         }
171         if ( maxJdkVersion == null && maxJavaMajorVersionNumber == -1 )
172         {
173             throw new IllegalArgumentException( "Exactly one of maxJdkVersion or "
174                 + "maxJavaMajorVersionNumber options should be set." );
175         }
176         if ( maxJdkVersion != null )
177         {
178             Integer needle = JDK_TO_MAJOR_VERSION_NUMBER_MAPPING.get( maxJdkVersion );
179             if ( needle == null )
180             {
181                 throw new IllegalArgumentException( "Unknown JDK version given. Should be something like \"1.7\"" );
182             }
183             maxJavaMajorVersionNumber = needle;
184         }
185         if ( maxJavaMajorVersionNumber == -1 )
186         {
187             throw new EnforcerRuleException( "maxJavaMajorVersionNumber must be set in the plugin configuration" );
188         }
189         if ( ignoreClasses != null )
190         {
191             IgnorableDependency ignorableDependency = new IgnorableDependency();
192             ignorableDependency.applyIgnoreClasses( ignoreClasses, false );
193             ignorableDependencies.add( ignorableDependency );
194         }
195     }

197     protected Set<Artifact> checkDependencies( Set<Artifact> dependencies, Log log )
198         throws EnforcerRuleException
199     {
200         long beforeCheck = System.currentTimeMillis();
201         Set<Artifact> problematic = new LinkedHashSet<Artifact>();
202         for ( Iterator<Artifact> it = dependencies.iterator(); it.hasNext(); )
203         {
204             Artifact artifact = it.next();
205             getLog().debug( "Analyzing artifact " + artifact );
206             String problem = isBadArtifact( artifact );
207             if ( problem != null )
208             {
209                 getLog().info( problem );
210                 problematic.add( artifact );
211             }
212         }
213         getLog().debug( "Bytecode version analysis took " + ( System.currentTimeMillis() - beforeCheck ) + " ms" );
214         return problematic;
215     }

217     private String isBadArtifact( Artifact a )
218         throws EnforcerRuleException
219     {
220         File f = a.getFile();
221         if ( !f.getName().endsWith( ".jar" ) )
222         {
223             return null;
224         }
225         try
226         {
227             JarFile jarFile = new JarFile( f );
228             try
229             {
230                 getLog().debug( f.getName() + " => " + f.getPath() );
231                 byte[] magicAndClassFileVersion = new byte[8];
232                 JAR:
233                 for ( Enumeration<JarEntry> e = jarFile.entries(); e.hasMoreElements(); )
234                 {
235                     JarEntry entry = e.nextElement();
236                     if ( !entry.isDirectory() && entry.getName().endsWith( ".class" ) )
237                     {
238                         for ( IgnorableDependency i : ignorableDependencies ) {
239                             if ( i.matches(entry.getName()))
240                             {
241                                 continue JAR;
242                             }
243                         }
244 
245                         StringBuilder builder = new StringBuilder();
246                         builder.append( "\t" ).append( entry.getName() ).append( " => " );
247                         InputStream is = null;
248                         try
249                         {
250                             is = jarFile.getInputStream( entry );
251                             int total = magicAndClassFileVersion.length;
252                             while ( total > 0 )
253                             {
254                                 int read = is.read( magicAndClassFileVersion, magicAndClassFileVersion.length - total,
255                                         total );
256 
257                                 if ( read == -1 )
258                                 {
259                                     throw new EOFException( f.toString() );
260                                 }
261 
262                                 total -= read;
263                             }
264 
265                             is.close();
266                             is = null;
267                         }
268                         finally
269                         {
270                             if ( is != null )
271                             {
272                                 is.close();
273                             }
274                         }
275 
276                         int minor = ( magicAndClassFileVersion[4] << 8 ) + magicAndClassFileVersion[5];
277                         int major = ( magicAndClassFileVersion[6] << 8 ) + magicAndClassFileVersion[7];
278                         builder.append( "major=" ).append( major ).append( ",minor=" ).append( minor );
279                         getLog().debug( builder.toString() );
280 
281                         if ( ( major > maxJavaMajorVersionNumber )
282                             || ( major == maxJavaMajorVersionNumber && minor > maxJavaMinorVersionNumber ) )
283                         {
284                             return "Restricted to " + renderVersion( maxJavaMajorVersionNumber, maxJavaMinorVersionNumber ) + " yet " + a + " contains " + entry.getName() + " targeted to " + renderVersion( major, minor );
285                         }
286                     }
287                 }
288             }
289             finally
290             {
291                 jarFile.close();
292             }
293         }
294         catch ( IOException e )
295         {
296             throw new EnforcerRuleException( "Error while reading jar file", e );
297         }
298         return null;
299     }

322     /*
323      * Filter the dependency artifacts according to the includes and excludes
324      * If includes and excludes are both null, the original set is returned.
325      *
326      * @param dependencies the list of dependencies to filter
327      * @return the resulting set of dependencies
328      */
329     private Set<Artifact> filterArtifacts( Set<Artifact> dependencies )
330     {
331         if ( includes == null && excludes == null && ignoredScopes == null )
332         {
333             return dependencies;
334         }
335 
336         AndArtifactFilter filter = new AndArtifactFilter( );
337         if ( includes != null )
338         {
339             filter.add( new StrictPatternIncludesArtifactFilter( includes ) );
340         }
341         if ( excludes != null )
342         {
343             filter.add( new StrictPatternExcludesArtifactFilter( excludes ) );
344         }
345 
346         Set<Artifact> result = new HashSet<Artifact>();
347         for ( Artifact artifact : dependencies )
348         {
349             if ( ignoredScopes != null && Arrays.asList( ignoredScopes ).contains( artifact.getScope() ) )
350             {
351                 continue;
352             }
353             if ( filter.include( artifact ) )
354             {
355                 result.add( artifact );
356             }
357         }
358         return result;
359     }
