42  /**
43   * This rule checks that certain properties are set and diverge from the ones given in another project.
44   *
45   * This allows to enforce that a property is e.g. overridden in a child project.
46   *
47   * @author Mirko Friedenhagen
48   * @since 1.0-alpha-3
49   */
50  public class RequirePropertyDiverges implements EnforcerRule
51  {
52      private String message;
53      
54      static final String MAVEN_ENFORCER_PLUGIN = "org.apache.maven.plugins:maven-enforcer-plugin";
55      /**
56       * Specify the required property. Must be given.
57       */
58      private String property = null;
59      /**
60       * Match the property value to a given regular expression. Defaults to value of defining project.
61       */
62      private String regex = null;
63      private final String ruleName = StringUtils.lowercaseFirstLetter( getClass().getSimpleName() );

65      /**
66       * Execute the rule.
67       *
68       * @param helper the helper
69       * @throws EnforcerRuleException the enforcer rule exception
70       */
71      public void execute( EnforcerRuleHelper helper ) throws EnforcerRuleException
72      {
73          final Log log = helper.getLog();
74  
75          Object propValue = getPropertyValue( helper );
76          checkPropValueNotBlank( propValue );
77  
78          final MavenProject project = getMavenProject( helper );
79          log.debug( getRuleName() + ": checking property '" + property + "' for project " + project );
80  
81          final MavenProject parent = findDefiningParent( project );
82  
83          // fail fast if the defining parent could not be found due to a bug in the rule
84          if ( parent == null )
85          {
86              throw new IllegalStateException( "Failed to find parent POM which defines the current rule" );
87          }
88  
89          if ( project.equals( parent ) )
90          {
91              log.debug( getRuleName() + ": skip for property '" + property + "' as " + project + " defines rule." );
92          }
93          else
94          {
95              log.debug( "Check configuration defined in " + parent );
96              if ( regex == null )
97              {
98                  checkAgainstParentValue( project, parent, helper, propValue );
99              }
100             else
101             {
102                 checkAgainstRegex( propValue );
103             }
104         }
105     }

107     /**
108      * Checks the value of the project against the one given in the defining ancestor project.
109      *
110      * @param project
111      * @param parent
112      * @param helper
113      * @param propValue
114      * @throws EnforcerRuleException
115      */
116     void checkAgainstParentValue( final MavenProject project, final MavenProject parent, EnforcerRuleHelper helper,
117             Object propValue ) throws EnforcerRuleException
118     {
119         final StringBuilder parentHierarchy = new StringBuilder( "project." );
120         MavenProject needle = project;
121         while ( !needle.equals( parent ) )
122         {
123             parentHierarchy.append( "parent." );
124             needle = needle.getParent();
125         }
126         final String propertyNameInParent = property.replace( "project.", parentHierarchy.toString() );
127         Object parentValue = getPropertyValue( helper, propertyNameInParent );
128         if ( propValue.equals( parentValue ) )
129         {
130             final String errorMessage = createResultingErrorMessage( String.format(
131                     "Property '%s' evaluates to '%s'. This does match '%s' from parent %s",
132                     property, propValue, parentValue, parent ) );
133             throw new EnforcerRuleException( errorMessage );
134         }
135     }

137     /**
138      * Checks the value of the project against the given regex.
139      *
140      *
141      * @param propValue
142      * @throws EnforcerRuleException
143      */
144     void checkAgainstRegex( Object propValue ) throws EnforcerRuleException
145     {
146         // Check that the property does not match the regex.
147         if ( propValue.toString().matches( regex ) )
148         {
149             final String errorMessage = createResultingErrorMessage(
150                     String.format(
151                     "Property '%s' evaluates to '%s'. This does match the regular expression '%s'",
152                     property, propValue, regex ) );
153             throw new EnforcerRuleException( errorMessage );
154         }
155     }

157     /**
158      * Finds the ancestor project which defines the rule.
159      *
160      * @param project to inspect
161      * @return the defining ancestor project.
162      */
163     final MavenProject findDefiningParent( final MavenProject project )
164     {
165         final Xpp3Dom invokingRule = createInvokingRuleDom();
166         MavenProject parent = project;
167         while ( parent != null )
168         {
169             final Model model = parent.getOriginalModel();
170             final Build build = model.getBuild();
171             if ( build != null )
172             {
173                 final List<Xpp3Dom> rules = getRuleConfigurations( build );
174                 if ( isDefiningProject( rules, invokingRule ) )
175                 {
176                     break;
177                 }
178             }
179             parent = parent.getParent();
180         }
181         return parent;
182     }

194     /**
195      * Checks whether ruleDom is in the list of rules from the model.
196      *
197      * @param rulesFromModel
198      * @param invokingRule
199      * @return true when the rules contain the invoking rule.
200      */
201     final boolean isDefiningProject( final List<Xpp3Dom> rulesFromModel, final Xpp3Dom invokingRule )
202     {
203         for ( final Xpp3Dom rule : rulesFromModel )
204         {
205             if ( rule.equals( invokingRule ) )
206             {
207                 return true;
208             }
209         }
210         return false;
211     }

223     /**
224      * Returns the rule configurations from the <tt>pluginManagement</tt> as well
225      * as the <tt>plugins</tt> section.
226      *
227      * @param build the build to inspect.
228      * @return configuration of the rules, may be an empty list.
229      */
230     final List<Xpp3Dom> getRuleConfigurations( final Build build )
231     {
232         @SuppressWarnings( "unchecked" )
233         final Map<String, Plugin> plugins = build.getPluginsAsMap();
234         final List<Xpp3Dom> ruleConfigurationsForPlugins = getRuleConfigurations( plugins );
235         final PluginManagement pluginManagement = build.getPluginManagement();
236         if ( pluginManagement != null )
237         {
238             @SuppressWarnings( "unchecked" )
239             final Map<String, Plugin> pluginsFromManagementAsMap = pluginManagement.getPluginsAsMap();
240             List<Xpp3Dom> ruleConfigurationsFromManagement = getRuleConfigurations( pluginsFromManagementAsMap );
241             ruleConfigurationsForPlugins.addAll( ruleConfigurationsFromManagement );
242         }
243         return ruleConfigurationsForPlugins;
244     }

246     /**
247      * Returns the list of <tt>requirePropertyDiverges</tt> configurations from the map of plugins.
248      *
249      * @param plugins
250      * @return list of requirePropertyDiverges configurations.
251      */
252     List<Xpp3Dom> getRuleConfigurations( final Map<String, Plugin> plugins )
253     {
254         if ( plugins.containsKey( MAVEN_ENFORCER_PLUGIN ) )
255         {
256             final List<Xpp3Dom> ruleConfigurations = new ArrayList<Xpp3Dom>();
257 
258             final Plugin enforcer = plugins.get( MAVEN_ENFORCER_PLUGIN );
259             final Xpp3Dom configuration = ( Xpp3Dom ) enforcer.getConfiguration();
260 
261             // add rules from plugin configuration
262             addRules( configuration, ruleConfigurations );
263 
264             // add rules from all plugin execution configurations
265             for ( Object execution : enforcer.getExecutions() )
266             {
267                 addRules( ( Xpp3Dom ) ( ( PluginExecution ) execution ).getConfiguration(), ruleConfigurations );
268             }
269 
270             return ruleConfigurations;
271         }
272         else
273         {
274             return new ArrayList();
275         }
276     }

278     /**
279      * Add the rules found in the given configuration to the list of rule configurations.
280      *
281      * @param configuration
282      *            configuration from which the rules are copied. May be <code>null</code>.
283      * @param ruleConfigurations
284      *            List to which the rules will be added.
285      */
286     private void addRules( final Xpp3Dom configuration, final List<Xpp3Dom> ruleConfigurations )
287     {
288         // may be null when rules are defined in pluginManagement during invocation
289         // for plugin section and vice versa.
290         if ( configuration != null )
291         {
292             final Xpp3Dom rules = configuration.getChild( "rules" );
293             if ( rules != null )
294             {
295                 final List<Xpp3Dom> originalListFromPom = Arrays.asList( rules.getChildren( getRuleName() ) );
296                 ruleConfigurations.addAll( createRuleListWithNameSortedChildren( originalListFromPom ) );
297             }
298         }
299     }

301     /**
302      * As Xpp3Dom is very picky about the order of children while comparing, create a new list where the children
303      * are added in alphabetical order. See <a href="https://jira.codehaus.org/browse/MOJO-1931">MOJO-1931</a>.
304      *
305      * @param originalListFromPom order not specified
306      * @return a list where children's member are alphabetically sorted.
307      */
308     private List<Xpp3Dom> createRuleListWithNameSortedChildren( final List<Xpp3Dom> originalListFromPom )
309     {
310         final List<Xpp3Dom> listWithSortedEntries = new ArrayList<Xpp3Dom>( originalListFromPom.size() );
311         for ( Xpp3Dom unsortedXpp3Dom : originalListFromPom )
312         {
313             final Xpp3Dom sortedXpp3Dom = new Xpp3Dom( getRuleName() );
314             final SortedMap<String, Xpp3Dom> childrenMap = new TreeMap<String, Xpp3Dom>();
315             final Xpp3Dom[] children = unsortedXpp3Dom.getChildren();
316             for ( Xpp3Dom child : children )
317             {
318                 childrenMap.put( child.getName(), child );
319             }
320             for ( Xpp3Dom entry : childrenMap.values() )
321             {
322                 sortedXpp3Dom.addChild( entry );
323             }
324             listWithSortedEntries.add( sortedXpp3Dom );
325         }
326         return listWithSortedEntries;
327     }

442     /**
443      * Creates the DOM of the invoking rule, but returns the children alphabetically sorted.
444      */
445     private static class CreateInvokingRuleDom
446     {
447 
448         private final Xpp3Dom ruleDom;
449         private final SortedMap<String, Xpp3Dom> map = new TreeMap<String, Xpp3Dom>();
450 
451         /** Real work is done in the constructor */
452         public CreateInvokingRuleDom( RequirePropertyDiverges rule )
453         {
454             ruleDom = new Xpp3Dom( rule.getRuleName() );
455             addToMapWhenNotNull( rule.property, "property" );
456             addToMapWhenNotNull( rule.message, "message" );
457             addToMapWhenNotNull( rule.regex, "regex" );
458             addChildrenToRuleDom();
459         }
460 
461         /**
462          * Readily prepared in constructor.
463          *
464          * @return the ruleDom
465          */
466         public Xpp3Dom getRuleDom() {
467             return ruleDom;
468         }
469 
470         private void addToMapWhenNotNull( String member, final String memberName )
471         {
472             if ( member != null )
473             {
474                 final Xpp3Dom memberDom = new Xpp3Dom( memberName );
475                 memberDom.setValue( member );
476                 map.put( memberName, memberDom );
477             }
478         }
479         
480         private void addChildrenToRuleDom()
481         {
482             for ( Xpp3Dom entry : map.values() )
483             {
484                 ruleDom.addChild( entry );
485             }
486         }
487     }
